                def get_string_array(verb, args)
                        csv = get_string(verb, args)
                        token = ""
                        tokens = []
                        csv.length.times do |i|
                                letter = csv[i,1]
                                if (letter == '\\')
                                        i++
                                        letter = csv[i,1]
                                        token = token + letter
                                elsif (letter == ',')
                                        tokens.push(token)
                                        token = ""
                                else
                                        token = token + letter
                                end
                        end
                        return tokens
                end
