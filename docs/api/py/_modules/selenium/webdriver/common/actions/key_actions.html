    def __init__(self, source: KeyInput | PointerInput | WheelInput | None = None) -> None:
        if not source:
            source = KeyInput(KEY)
        self.source = source
        super().__init__(source)

[docs]    def send_keys(self, text: str | list) -> KeyActions:
        if not isinstance(text, list):
            text = keys_to_typing(text)
        for letter in text:
            self.key_down(letter)
            self.key_up(letter)
        return self


    def _key_action(self, action: str, letter) -> KeyActions:
        meth = getattr(self.source, action)
        meth(letter)
        return self
