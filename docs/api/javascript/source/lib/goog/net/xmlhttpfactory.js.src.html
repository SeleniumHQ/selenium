50/**
51 * @return {Object} Options describing how xhr objects obtained from this
52 * factory should be used.
53 */
54goog.net.XmlHttpFactory.prototype.getOptions = function() {

56 (this.cachedOptions_ = this.internalGetOptions());
57};
