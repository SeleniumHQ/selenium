198bot.events.EventFactory_ = function(type, bubbles, cancelable) {

200  this.type_ = type;

210/**
211 * Creates an event.
212 *
213 * @param {!Element|!Window} target Target element of the event.
214 * @param {bot.events.EventArgs=} opt_args Event arguments.
215 * @return {!Event} Newly created event.
216 */
217bot.events.EventFactory_.prototype.create = function(target, opt_args) {
218  var doc = goog.dom.getOwnerDocument(target);
219  var event;
220
221  if (bot.userAgent.IE_DOC_PRE9) {
222    event = doc.createEventObject();
223  } else {
224    event = doc.createEvent('HTMLEvents');
225    event.initEvent(this.type_, this.bubbles_, this.cancelable_);
226  }
227
228  return event;
229};

260/**
261 * @override
262 */
263bot.events.MouseEventFactory_.prototype.create = function(target, opt_args) {
264  // Only Gecko supports the mouse pixel scroll event.
265  if (!goog.userAgent.GECKO && this == bot.events.EventType.MOUSEPIXELSCROLL) {
266    throw new bot.Error(bot.ErrorCode.UNSUPPORTED_OPERATION,
267        'Browser does not support a mouse pixel scroll event.');
268  }
269
270  var args = /** @type {!bot.events.MouseArgs} */ (opt_args);
271  var doc = goog.dom.getOwnerDocument(target);
272  var event;
273
274  if (bot.userAgent.IE_DOC_PRE9) {
275    event = doc.createEventObject();
276    event.altKey = args.altKey;
277    event.ctrlKey = args.ctrlKey;
278    event.metaKey = args.metaKey;
279    event.shiftKey = args.shiftKey;
280    event.button = args.button;
281
282    // NOTE: ie8 does a strange thing with the coordinates passed in the event:
283    // - if offset{X,Y} coordinates are specified, they are also used for
284    //   client{X,Y}, event if client{X,Y} are also specified.
285    // - if only client{X,Y} are specified, they are also used for offset{x,y}
286    // Thus, for ie8, it is impossible to set both offset and client
287    // and have them be correct when they come out on the other side.
288    event.clientX = args.clientX;
289    event.clientY = args.clientY;
290
291    // Sets a property of the event object using Object.defineProperty.
292    // Some readonly properties of the IE event object can only be set this way.
293    var setEventProperty = function(prop, value) {
294      Object.defineProperty(event, prop, {
295        get: function() {
296          return value;
297        }
298      });
299    };
300
301    // IE has fromElement and toElement properties, no relatedTarget property.
302    // IE does not allow fromElement and toElement to be set directly, but
303    // Object.defineProperty can redefine them, when it is available. Do not
304    // use Object.defineProperties (plural) because it is even less supported.
305    // If defineProperty is unavailable, fall back to setting the relatedTarget,
306    // which many event frameworks, including jQuery and Closure, forgivingly
307    // pass on as the relatedTarget on their event object abstraction.
308    if (this == bot.events.EventType.MOUSEOUT ||
309        this == bot.events.EventType.MOUSEOVER) {
310      if (Object.defineProperty) {
311        var out = (this == bot.events.EventType.MOUSEOUT);
312        setEventProperty('fromElement', out ? target : args.relatedTarget);
313        setEventProperty('toElement', out ? args.relatedTarget : target);
314      } else {
315        event.relatedTarget = args.relatedTarget;
316      }
317    }
318
319    // IE does not allow the wheelDelta property to be set directly, so we can
320    // only do it where defineProperty is supported; otherwise store the wheel
321    // delta in the event "detail" as a last resort in case the app looks there.
322    if (this == bot.events.EventType.MOUSEWHEEL) {
323      if (Object.defineProperty) {
324        setEventProperty('wheelDelta', args.wheelDelta);
325      } else {
326        event.detail = args.wheelDelta;
327      }
328    }
329  } else {
330    var view = goog.dom.getWindow(doc);
331    event = doc.createEvent('MouseEvents');
332    var detail = 1;
333
334    // All browser but Firefox provide the wheelDelta value in the event.
335    // Firefox provides the scroll amount in the detail field, where it has the
336    // opposite polarity of the wheelDelta (upward scroll is negative) and is a
337    // factor of 40 less than the wheelDelta value. Opera provides both values.
338    // The wheelDelta value is normally some multiple of 40.
339    if (this == bot.events.EventType.MOUSEWHEEL) {
340      if (!goog.userAgent.GECKO) {
341        event.wheelDelta = args.wheelDelta;
342      }
343      if (goog.userAgent.GECKO || goog.userAgent.OPERA) {
344        detail = args.wheelDelta / -40;
345      }
346    }
347
348    // Only Gecko supports a mouse pixel scroll event, so we use it as the
349    // "standard" and pass it along as is as the "detail" of the event.
350    if (goog.userAgent.GECKO && this == bot.events.EventType.MOUSEPIXELSCROLL) {
351      detail = args.wheelDelta;
352    }
353
354    event.initMouseEvent(this.type_, this.bubbles_, this.cancelable_, view,
355        detail, /*screenX*/ 0, /*screenY*/ 0, args.clientX, args.clientY,
356        args.ctrlKey, args.altKey, args.shiftKey, args.metaKey, args.button,
357        args.relatedTarget);
358
359    // Trying to modify the properties throws an error,
360    // so we define getters to return the correct values.
361    if (goog.userAgent.IE &&
362        event.pageX === 0 && event.pageY === 0 && Object.defineProperty) {
363      var scrollElem = goog.dom.getDomHelper(target).getDocumentScrollElement();
364      var clientElem = goog.style.getClientViewportElement(doc);
365      var pageX = args.clientX + scrollElem.scrollLeft - clientElem.clientLeft;
366      var pageY = args.clientY + scrollElem.scrollTop - clientElem.clientTop;
367
368      Object.defineProperty(event, 'pageX', {
369        get: function() {
370          return pageX;
371        }
372      });
373      Object.defineProperty(event, 'pageY', {
374        get: function() {
375          return pageY;
376        }
377      });
378    }
379  }
380
381  return event;
382};

402/**
403 * @override
404 */
405bot.events.KeyboardEventFactory_.prototype.create = function(target, opt_args) {
406  var args = /** @type {!bot.events.KeyboardArgs} */ (opt_args);
407  var doc = goog.dom.getOwnerDocument(target);
408  var event;
409
410  if (goog.userAgent.GECKO) {
411    var view = goog.dom.getWindow(doc);
412    var keyCode = args.charCode ? 0 : args.keyCode;
413    event = doc.createEvent('KeyboardEvent');
414    event.initKeyEvent(this.type_, this.bubbles_, this.cancelable_, view,
415        args.ctrlKey, args.altKey, args.shiftKey, args.metaKey, keyCode,
416        args.charCode);
417    // https://bugzilla.mozilla.org/show_bug.cgi?id=501496
418    if (this.type_ == bot.events.EventType.KEYPRESS && args.preventDefault) {
419      event.preventDefault();
420    }
421  } else {
422    if (bot.userAgent.IE_DOC_PRE9) {
423      event = doc.createEventObject();
424    } else {  // WebKit, Opera, and IE 9+ in Standards mode.
425      event = doc.createEvent('Events');
426      event.initEvent(this.type_, this.bubbles_, this.cancelable_);
427    }
428    event.altKey = args.altKey;
429    event.ctrlKey = args.ctrlKey;
430    event.metaKey = args.metaKey;
431    event.shiftKey = args.shiftKey;
432    event.keyCode = args.charCode || args.keyCode;
433    if (goog.userAgent.WEBKIT) {
434      event.charCode = (this == bot.events.EventType.KEYPRESS) ?
435          event.keyCode : 0;
436    }
437  }
438
439  return event;
440};

460/**
461 * @override
462 */
463bot.events.TouchEventFactory_.prototype.create = function(target, opt_args) {
464  if (!bot.events.SUPPORTS_TOUCH_EVENTS) {
465    throw new bot.Error(bot.ErrorCode.UNSUPPORTED_OPERATION,
466        'Browser does not support firing touch events.');
467  }
468
469  var args = /** @type {!bot.events.TouchArgs} */ (opt_args);
470  var doc = goog.dom.getOwnerDocument(target);
471  var view = goog.dom.getWindow(doc);
472
473  // Creates a TouchList, using native touch Api, for touch events.
474  function createNativeTouchList(touchListArgs) {
475    var touches = goog.array.map(touchListArgs, function(touchArg) {
476      return doc.createTouch(view, target, touchArg.identifier,
477          touchArg.pageX, touchArg.pageY, touchArg.screenX, touchArg.screenY);
478    });
479
480    return doc.createTouchList.apply(doc, touches);
481  }
482
483  // Creates a TouchList, using simulated touch Api, for touch events.
484  function createGenericTouchList(touchListArgs) {
485    var touches = goog.array.map(touchListArgs, function(touchArg) {
486      // The target field is not part of the W3C spec, but both android and iOS
487      // add the target field to each touch.
488      return {
489        identifier: touchArg.identifier,
490        screenX: touchArg.screenX,
491        screenY: touchArg.screenY,
492        clientX: touchArg.clientX,
493        clientY: touchArg.clientY,
494        pageX: touchArg.pageX,
495        pageY: touchArg.pageY,
496        target: target
497      };
498    });
499    touches.item = function(i) {
500      return touches[i];
501    };
502    return touches;
503  }
504
505  function createTouchList(touches) {
506    return bot.events.BROKEN_TOUCH_API_ ?
507        createGenericTouchList(touches) :
508        createNativeTouchList(touches);
509  }
510
511  // As a performance optimization, reuse the created touchlist when the lists
512  // are the same, which is often the case in practice.
513  var changedTouches = createTouchList(args.changedTouches);
514  var touches = (args.touches == args.changedTouches) ?
515      changedTouches : createTouchList(args.touches);
516  var targetTouches = (args.targetTouches == args.changedTouches) ?
517      changedTouches : createTouchList(args.targetTouches);
518
519  var event;
520  if (bot.events.BROKEN_TOUCH_API_) {
521    event = doc.createEvent('MouseEvents');
522    event.initMouseEvent(this.type_, this.bubbles_, this.cancelable_, view,
523        /*detail*/ 1, /*screenX*/ 0, /*screenY*/ 0, args.clientX, args.clientY,
524        args.ctrlKey, args.altKey, args.shiftKey, args.metaKey, /*button*/ 0,
525        args.relatedTarget);
526    event.touches = touches;
527    event.targetTouches = targetTouches;
528    event.changedTouches = changedTouches;
529    event.scale = args.scale;
530    event.rotation = args.rotation;
531  } else {
532    event = doc.createEvent('TouchEvent');
533    if (goog.userAgent.product.ANDROID) {
534      // Android's initTouchEvent method is not compliant with the W3C spec.
535      event.initTouchEvent(touches, targetTouches, changedTouches,
536          this.type_, view, /*screenX*/ 0, /*screenY*/ 0, args.clientX,
537          args.clientY, args.ctrlKey, args.altKey, args.shiftKey, args.metaKey);
538    } else {
539      event.initTouchEvent(this.type_, this.bubbles_, this.cancelable_, view,
540          /*detail*/ 1, /*screenX*/ 0, /*screenY*/ 0, args.clientX,
541          args.clientY, args.ctrlKey, args.altKey, args.shiftKey, args.metaKey,
542          touches, targetTouches, changedTouches, args.scale, args.rotation);
543    }
544    event.relatedTarget = args.relatedTarget;
545  }
546
547  return event;
548};

725/**
726 * Fire a named event on a particular element.
727 *
728 * @param {!Element|!Window} target The element on which to fire the event.
729 * @param {!bot.events.EventType} type Event type.
730 * @param {bot.events.EventArgs=} opt_args Arguments to initialize the event.
731 * @return {boolean} Whether the event fired successfully or was cancelled.
732 */
733bot.events.fire = function(target, type, opt_args) {
734  var factory = /** @type {!bot.events.EventFactory_} */ (type);
735  var event = factory.create(target, opt_args);
736
737  // Ensure the event's isTrusted property is set to false, so that
738  // bot.events.isSynthetic() can identify synthetic events from native ones.
739  if (!('isTrusted' in event)) {
740    event['isTrusted'] = false;
741  }
742
743  if (bot.userAgent.IE_DOC_PRE9) {
744    return target.fireEvent('on' + factory.type_, event);
745  } else {
746    return target.dispatchEvent(event);
747  }
748};

751/**
752 * Returns whether the event was synthetically created by the atoms;
753 * if false, was created by the browser in response to a live user action.
754 *
755 * @param {!(Event|goog.events.BrowserEvent)} event An event.
756 * @return {boolean} Whether the event was synthetically created.
757 */
758bot.events.isSynthetic = function(event) {
759  var e = event.getBrowserEvent ? event.getBrowserEvent() : event;
760  return 'isTrusted' in e ? !e['isTrusted'] : false;
