39/**
40 * Retrieves the active element for a node's owner document.
41 * @param {!(Node|Window)} nodeOrWindow The node whose owner document to get
42 *     the active element for.
43 * @return {Element} The active element, if any.
44 */
45bot.dom.getActiveElement = function(nodeOrWindow) {
46  var active = goog.dom.getActiveElement(
47      goog.dom.getOwnerDocument(nodeOrWindow));
48  // IE has the habit of returning an empty object from
49  // goog.dom.getActiveElement instead of null.
50  if (goog.userAgent.IE &&
51      active &&
52      typeof active.nodeType === 'undefined') {
53    return null;
54  }
55  return active;
56};

59/**
60 * Returns whether the given node is an element and, optionally, whether it has
61 * the given tag name. If the tag name is not provided, returns true if the node
62 * is an element, regardless of the tag name.h
63 *
64 * @param {Node} node The node to test.
65 * @param {string=} opt_tagName Tag name to test the node for.
66 * @return {boolean} Whether the node is an element with the given tag name.
67 */
68bot.dom.isElement = function(node, opt_tagName) {
69  return !!node && node.nodeType == goog.dom.NodeType.ELEMENT &&
70      (!opt_tagName || node.tagName.toUpperCase() == opt_tagName);

106/**
107 * Returns whether the element can be checked or selected.
108 *
109 * @param {!Element} element The element to check.
110 * @return {boolean} Whether the element could be checked or selected.
111 */
112bot.dom.isSelectable = function(element) {
113  if (bot.dom.isElement(element, goog.dom.TagName.OPTION)) {
114    return true;
115  }
116
117  if (bot.dom.isElement(element, goog.dom.TagName.INPUT)) {
118    var type = element.type.toLowerCase();
119    return type == 'checkbox' || type == 'radio';
120  }
121
122  return false;
123};

126/**
127 * Returns whether the element is checked or selected.
128 *
129 * @param {!Element} element The element to check.
130 * @return {boolean} Whether the element is checked or selected.
131 */
132bot.dom.isSelected = function(element) {
133  if (!bot.dom.isSelectable(element)) {
134    throw new bot.Error(bot.ErrorCode.ELEMENT_NOT_SELECTABLE,
135        'Element is not selectable');
136  }
137
138  var propertyName = 'selected';
139  var type = element.type && element.type.toLowerCase();
140  if ('checkbox' == type || 'radio' == type) {
141    propertyName = 'checked';
142  }
143
144  return !!bot.dom.getProperty(element, propertyName);
145};

179/**
180 * Looks up the given property (not to be confused with an attribute) on the
181 * given element.
182 *
183 * @param {!Element} element The element to use.
184 * @param {string} propertyName The name of the property.
185 * @return {*} The value of the property.
186 */
187bot.dom.getProperty = function(element, propertyName) {
188  // When an <option>'s value attribute is not set, its value property should be
189  // its text content, but IE < 8 does not adhere to that behavior, so fix it.
190  // http://www.w3.org/TR/1999/REC-html401-19991224/interact/forms.html#adef-value-OPTION
191  if (bot.userAgent.IE_DOC_PRE8 && propertyName == 'value' &&
192      bot.dom.isElement(element, goog.dom.TagName.OPTION) &&
193      goog.isNull(bot.dom.getAttribute(element, 'value'))) {
194    return goog.dom.getRawTextContent(element);
195  }
196  return element[propertyName];
197};

200/**
201 * Regex to split on semicolons, but not when enclosed in parens or quotes.
202 * Helper for {@link bot.dom.standardizeStyleAttribute_}.
203 * If the style attribute ends with a semicolon this will include an empty
204 * string at the end of the array
205 * @private {!RegExp}
206 * @const
207 */
208bot.dom.SPLIT_STYLE_ATTRIBUTE_ON_SEMICOLONS_REGEXP_ =
209    new RegExp('[;]+' +
210               '(?=(?:(?:[^"]*"){2})*[^"]*$)' +
211               '(?=(?:(?:[^\']*\'){2})*[^\']*$)' +
212               '(?=(?:[^()]*\\([^()]*\\))*[^()]*$)');
213
214
215/**
216 * Standardize a style attribute value, which includes:
217 *  (1) converting all property names lowercase
218 *  (2) ensuring it ends in a trailing semi-colon
219 *  (3) removing empty style values (which only appear on Opera).
220 * @param {string} value The style attribute value.
221 * @return {string} The identical value, with the formatting rules described
222 *     above applied.
223 * @private
224 */
225bot.dom.standardizeStyleAttribute_ = function(value) {
226  var styleArray = value.split(
227      bot.dom.SPLIT_STYLE_ATTRIBUTE_ON_SEMICOLONS_REGEXP_);
228  var css = [];
229  goog.array.forEach(styleArray, function(pair) {
230    var i = pair.indexOf(':');
231    if (i > 0) {
232      var keyValue = [pair.slice(0, i), pair.slice(i + 1)];
233      if (keyValue.length == 2) {
234        css.push(keyValue[0].toLowerCase(), ':', keyValue[1], ';');
235      }
236    }
237  });
238  css = css.join('');
239  css = css.charAt(css.length - 1) == ';' ? css : css + ';';
240  return goog.userAgent.OPERA ? css.replace(/\w+:;/g, '') : css;
241};

244/**
245 * Get the user-specified value of the given attribute of the element, or null
246 * if the attribute is not present.
247 *
248 * <p>For boolean attributes such as "selected" or "checked", this method
249 * returns the value of element.getAttribute(attributeName) cast to a String
250 * when attribute is present. For modern browsers, this will be the string the
251 * attribute is given in the HTML, but for IE8 it will be the name of the
252 * attribute, and for IE7, it will be the string "true". To test whether a
253 * boolean attribute is present, test whether the return value is non-null, the
254 * same as one would for non-boolean attributes. Specifically, do *not* test
255 * whether the boolean evaluation of the return value is true, because the value
256 * of a boolean attribute that is present will often be the empty string.
257 *
258 * <p>For the style attribute, it standardizes the value by lower-casing the
259 * property names and always including a trailing semi-colon.
260 *
261 * @param {!Element} element The element to use.
262 * @param {string} attributeName The name of the attribute to return.
263 * @return {?string} The value of the attribute or "null" if entirely missing.
264 */
265bot.dom.getAttribute = function(element, attributeName) {
266  attributeName = attributeName.toLowerCase();
267
268  // The style attribute should be a css text string that includes only what
269  // the HTML element specifies itself (excluding what is inherited from parent
270  // elements or style sheets). We standardize the format of this string via the
271  // standardizeStyleAttribute method.
272  if (attributeName == 'style') {
273    return bot.dom.standardizeStyleAttribute_(element.style.cssText);
274  }
275
276  // In IE doc mode < 8, the "value" attribute of an <input> is only accessible
277  // as a property.
278  if (bot.userAgent.IE_DOC_PRE8 && attributeName == 'value' &&
279      bot.dom.isElement(element, goog.dom.TagName.INPUT)) {
280    return element['value'];
281  }
282
283  // In IE < 9, element.getAttributeNode will return null for some boolean
284  // attributes that are present, such as the selected attribute on <option>
285  // elements. This if-statement is sufficient if these cases are restricted
286  // to boolean attributes whose reflected property names are all lowercase
287  // (as attributeName is by this point), like "selected". We have not
288  // found a boolean attribute for which this does not work.
289  if (bot.userAgent.IE_DOC_PRE9 && element[attributeName] === true) {
290    return String(element.getAttribute(attributeName));
291  }
292
293  // When the attribute is not present, either attr will be null or
294  // attr.specified will be false.
295  var attr = element.getAttributeNode(attributeName);
296  return (attr && attr.specified) ? attr.value : null;
297};

317/**
318 * Determines if an element is enabled. An element is considered enabled if it
319 * does not support the "disabled" attribute, or if it is not disabled.
320 * @param {!Element} el The element to test.
321 * @return {boolean} Whether the element is enabled.
322 */
323bot.dom.isEnabled = function(el) {
324  var tagName = el.tagName.toUpperCase();
325  if (!goog.array.contains(bot.dom.DISABLED_ATTRIBUTE_SUPPORTED_, tagName)) {
326    return true;
327  }
328
329  if (bot.dom.getProperty(el, 'disabled')) {
330    return false;
331  }
332
333  // The element is not explicitly disabled, but if it is an OPTION or OPTGROUP,
334  // we must test if it inherits its state from a parent.
335  if (el.parentNode &&
336      el.parentNode.nodeType == goog.dom.NodeType.ELEMENT &&
337      goog.dom.TagName.OPTGROUP == tagName ||
338      goog.dom.TagName.OPTION == tagName) {
339    return bot.dom.isEnabled(/**@type{!Element}*/ (el.parentNode));
340  }
341
342  // Is there an ancestor of the current element that is a disabled fieldset
343  // and whose child is also an ancestor-or-self of the current element but is
344  // not the first legend child of the fieldset. If so then the element is
345  // disabled.
346  return !goog.dom.getAncestor(el, function(e) {
347    var parent = e.parentNode;
348
349    if (parent &&
350        bot.dom.isElement(parent, goog.dom.TagName.FIELDSET) &&
351        bot.dom.getProperty(/** @type {!Element} */ (parent), 'disabled')) {
352      if (!bot.dom.isElement(e, goog.dom.TagName.LEGEND)) {
353        return true;
354      }
355
356      var sibling = e;
357      // Are there any previous legend siblings? If so then we are not the
358      // first and the element is disabled
359      while (sibling = goog.dom.getPreviousElementSibling(sibling)) {
360        if (bot.dom.isElement(sibling, goog.dom.TagName.LEGEND)) {
361          return true;
362        }
363      }
364    }
365    return false;
366  }, true);
367};

411/**
412 * @param {!Element} element The element to check.
413 * @return {boolean} Whether the element is contentEditable.
414 */
415bot.dom.isContentEditable = function(element) {
416  // Check if browser supports contentEditable.
417  if (!goog.isDef(element['contentEditable'])) {
418    return false;
419  }
420
421  // Checking the element's isContentEditable property is preferred except for
422  // IE where that property is not reliable on IE versions 7, 8, and 9.
423  if (!goog.userAgent.IE && goog.isDef(element['isContentEditable'])) {
424    return element.isContentEditable;
425  }
426
427  // For IE and for browsers where contentEditable is supported but
428  // isContentEditable is not, traverse up the ancestors:
429  function legacyIsContentEditable(e) {
430    if (e.contentEditable == 'inherit') {
431      var parent = bot.dom.getParentElement(e);
432      return parent ? legacyIsContentEditable(parent) : false;
433    } else {
434      return e.contentEditable == 'true';
435    }
436  }
437  return legacyIsContentEditable(element);
438};

457/**
458 * Returns the parent element of the given node, or null. This is required
459 * because the parent node may not be another element.
460 *
461 * @param {!Node} node The node who's parent is desired.
462 * @return {Element} The parent element, if available, null otherwise.
463 */
464bot.dom.getParentElement = function(node) {
465  var elem = node.parentNode;
466
467  while (elem &&
468         elem.nodeType != goog.dom.NodeType.ELEMENT &&
469         elem.nodeType != goog.dom.NodeType.DOCUMENT &&
470         elem.nodeType != goog.dom.NodeType.DOCUMENT_FRAGMENT) {
471    elem = elem.parentNode;
472  }
473  return /** @type {Element} */ (bot.dom.isElement(elem) ? elem : null);
474};

491/**
492 * Retrieves the implicitly-set, effective style of an element, or null if it is
493 * unknown. It returns the computed style where available; otherwise it looks
494 * up the DOM tree for the first style value not equal to 'inherit,' using the
495 * IE currentStyle of each node if available, and otherwise the inline style.
496 * Since the computed, current, and inline styles can be different, the return
497 * value of this function is not always consistent across browsers. See:
498 * http://code.google.com/p/doctype/wiki/ArticleComputedStyleVsCascadedStyle
499 *
500 * @param {!Element} elem Element to get the style value from.
501 * @param {string} propertyName Name of the CSS property in selector-case.
502 * @return {?string} The value of the style property, or null.
503 */
504bot.dom.getEffectiveStyle = function(elem, propertyName) {
505  var styleName = goog.string.toCamelCase(propertyName);
506  if (styleName == 'float' ||
507      styleName == 'cssFloat' ||
508      styleName == 'styleFloat') {
509    styleName = bot.userAgent.IE_DOC_PRE9 ? 'styleFloat' : 'cssFloat';
510  }
511  var style = goog.style.getComputedStyle(elem, styleName) ||
512      bot.dom.getCascadedStyle_(elem, styleName);
513  if (style === null) {
514    return null;
515  }
516  return bot.color.standardizeColor(propertyName, style);
517};

520/**
521 * Looks up the DOM tree for the first style value not equal to 'inherit,' using
522 * the currentStyle of each node if available, and otherwise the inline style.
523 *
524 * @param {!Element} elem Element to get the style value from.
525 * @param {string} styleName CSS style property in camelCase.
526 * @return {?string} The value of the style property, or null.
527 * @private
528 */
529bot.dom.getCascadedStyle_ = function(elem, styleName) {
530  var style = elem.currentStyle || elem.style;
531  var value = style[styleName];
532  if (!goog.isDef(value) && goog.isFunction(style['getPropertyValue'])) {
533    value = style['getPropertyValue'](styleName);
534  }
535
536  if (value != 'inherit') {
537    return goog.isDef(value) ? value : null;
538  }
539  var parent = bot.dom.getParentElement(elem);
540  return parent ? bot.dom.getCascadedStyle_(parent, styleName) : null;
541};

544/**
545 * Determines whether an element is what a user would call "shown". This means
546 * that the element is shown in the viewport of the browser, and only has
547 * height and width greater than 0px, and that its visibility is not "hidden"
548 * and its display property is not "none".
549 * Options and Optgroup elements are treated as special cases: they are
550 * considered shown iff they have a enclosing select element that is shown.
551 *
552 * @param {!Element} elem The element to consider.
553 * @param {boolean=} opt_ignoreOpacity Whether to ignore the element's opacity
554 *     when determining whether it is shown; defaults to false.
555 * @return {boolean} Whether or not the element is visible.
556 */
557bot.dom.isShown = function(elem, opt_ignoreOpacity) {
558  if (!bot.dom.isElement(elem)) {
559    throw new Error('Argument to isShown must be of type Element');
560  }
561
562  // Option or optgroup is shown iff enclosing select is shown (ignoring the
563  // select's opacity).
564  if (bot.dom.isElement(elem, goog.dom.TagName.OPTION) ||
565      bot.dom.isElement(elem, goog.dom.TagName.OPTGROUP)) {
566    var select = /**@type {Element}*/ (goog.dom.getAncestor(elem, function(e) {
567      return bot.dom.isElement(e, goog.dom.TagName.SELECT);
568    }));
569    return !!select && bot.dom.isShown(select, /*ignoreOpacity=*/true);
570  }
571
572  // Image map elements are shown if image that uses it is shown, and
573  // the area of the element is positive.
574  var imageMap = bot.dom.maybeFindImageMap_(elem);
575  if (imageMap) {
576    return !!imageMap.image &&
577           imageMap.rect.width > 0 && imageMap.rect.height > 0 &&
578           bot.dom.isShown(imageMap.image, opt_ignoreOpacity);
579  }
580
581  // Any hidden input is not shown.
582  if (bot.dom.isElement(elem, goog.dom.TagName.INPUT) &&
583      elem.type.toLowerCase() == 'hidden') {
584    return false;
585  }
586
587  // Any NOSCRIPT element is not shown.
588  if (bot.dom.isElement(elem, goog.dom.TagName.NOSCRIPT)) {
589    return false;
590  }
591
592  // Any element with hidden visibility is not shown.
593  if (bot.dom.getEffectiveStyle(elem, 'visibility') == 'hidden') {
594    return false;
595  }
596
597  // Any element with a display style equal to 'none' or that has an ancestor
598  // with display style equal to 'none' is not shown.
599  function displayed(e) {
600    if (bot.dom.getEffectiveStyle(e, 'display') == 'none') {
601      return false;
602    }
603    var parent = bot.dom.getParentElement(e);
604    return !parent || displayed(parent);
605  }
606  if (!displayed(elem)) {
607    return false;
608  }
609
610  // Any transparent element is not shown.
611  if (!opt_ignoreOpacity && bot.dom.getOpacity(elem) == 0) {
612    return false;
613  }
614
615  // Any element without positive size dimensions is not shown.
616  function positiveSize(e) {
617    var rect = bot.dom.getClientRect(e);
618    if (rect.height > 0 && rect.width > 0) {
619      return true;
620    }
621    // A vertical or horizontal SVG Path element will report zero width or
622    // height but is "shown" if it has a positive stroke-width.
623    if (bot.dom.isElement(e, 'PATH') && (rect.height > 0 || rect.width > 0)) {
624      var strokeWidth = bot.dom.getEffectiveStyle(e, 'stroke-width');
625      return !!strokeWidth && (parseInt(strokeWidth, 10) > 0);
626    }
627    // Zero-sized elements should still be considered to have positive size
628    // if they have a child element or text node with positive size, unless
629    // the element has an 'overflow' style of 'hidden'.
630    return bot.dom.getEffectiveStyle(e, 'overflow') != 'hidden' &&
631        goog.array.some(e.childNodes, function(n) {
632          return n.nodeType == goog.dom.NodeType.TEXT ||
633                 (bot.dom.isElement(n) && positiveSize(n));
634        });
635  }
636  if (!positiveSize(elem)) {
637    return false;
638  }
639
640  // Elements that are hidden by overflow are not shown.
641  function hiddenByOverflow(e) {
642    return bot.dom.getOverflowState(e) == bot.dom.OverflowState.HIDDEN &&
643        goog.array.every(e.childNodes, function(n) {
644          return !bot.dom.isElement(n) || hiddenByOverflow(n);
645        });
646  }
647  return !hiddenByOverflow(elem);
648};

651/**
652 * The kind of overflow area in which an element may be located. NONE if it does
653 * not overflow any ancestor element; HIDDEN if it overflows and cannot be
654 * scrolled into view; SCROLL if it overflows but can be scrolled into view.
655 *
656 * @enum {string}
657 */
658bot.dom.OverflowState = {
659  NONE: 'none',
660  HIDDEN: 'hidden',
661  SCROLL: 'scroll'
662};

665/**
666 * Returns the overflow state of the given element.
667 *
668 * If an optional coordinate or rectangle region is provided, returns the
669 * overflow state of that region relative to the element. A coordinate is
670 * treated as a 1x1 rectangle whose top-left corner is the coordinate.
671 *
672 * @param {!Element} elem Element.
673 * @param {!(goog.math.Coordinate|goog.math.Rect)=} opt_region
674 *     Coordinate or rectangle relative to the top-left corner of the element.
675 * @return {bot.dom.OverflowState} Overflow state of the element.
676 */
677bot.dom.getOverflowState = function(elem, opt_region) {
678  var region = bot.dom.getClientRegion(elem, opt_region);
679  var ownerDoc = goog.dom.getOwnerDocument(elem);
680  var htmlElem = ownerDoc.documentElement;
681  var bodyElem = ownerDoc.body;
682  var htmlOverflowStyle = bot.dom.getEffectiveStyle(htmlElem, 'overflow');
683  var treatAsFixedPosition;
684
685  // Return the closest ancestor that the given element may overflow.
686  function getOverflowParent(e) {
687    var position = bot.dom.getEffectiveStyle(e, 'position');
688    if (position == 'fixed') {
689      treatAsFixedPosition = true;
690      // Fixed-position element may only overflow the viewport.
691      return htmlElem;
692    } else {
693      var parent = bot.dom.getParentElement(e);
694      while (parent && !canBeOverflowed(parent)) {
695        parent = bot.dom.getParentElement(parent);
696      }
697      return parent;
698    }
699
700    function canBeOverflowed(container) {
701      // The HTML element can always be overflowed.
702      if (container == htmlElem) {
703        return true;
704      }
705      // An element cannot overflow an element with an inline display style.
706      var containerDisplay = /** @type {string} */ (
707          bot.dom.getEffectiveStyle(container, 'display'));
708      if (goog.string.startsWith(containerDisplay, 'inline')) {
709        return false;
710      }
711      // An absolute-positioned element cannot overflow a static-positioned one.
712      if (position == 'absolute' &&
713          bot.dom.getEffectiveStyle(container, 'position') == 'static') {
714        return false;
715      }
716      return true;
717    }
718  }
719
720  // Return the x and y overflow styles for the given element.
721  function getOverflowStyles(e) {
722    // When the <html> element has an overflow style of 'visible', it assumes
723    // the overflow style of the body, and the body is really overflow:visible.
724    var overflowElem = e;
725    if (htmlOverflowStyle == 'visible') {
726      // Note: bodyElem will be null/undefined in SVG documents.
727      if (e == htmlElem && bodyElem) {
728        overflowElem = bodyElem;
729      } else if (e == bodyElem) {
730        return {x: 'visible', y: 'visible'};
731      }
732    }
733    var overflow = {
734      x: bot.dom.getEffectiveStyle(overflowElem, 'overflow-x'),
735      y: bot.dom.getEffectiveStyle(overflowElem, 'overflow-y')
736    };
737    // The <html> element cannot have a genuine 'visible' overflow style,
738    // because the viewport can't expand; 'visible' is really 'auto'.
739    if (e == htmlElem) {
740      overflow.x = overflow.x == 'visible' ? 'auto' : overflow.x;
741      overflow.y = overflow.y == 'visible' ? 'auto' : overflow.y;
742    }
743    return overflow;
744  }
745
746  // Returns the scroll offset of the given element.
747  function getScroll(e) {
748    if (e == htmlElem) {
749      return new goog.dom.DomHelper(ownerDoc).getDocumentScroll();
750    } else {
751      return new goog.math.Coordinate(e.scrollLeft, e.scrollTop);
752    }
753  }
754
755  // Check if the element overflows any ancestor element.
756  for (var container = getOverflowParent(elem);
757       !!container;
758       container = getOverflowParent(container)) {
759    var containerOverflow = getOverflowStyles(container);
760
761    // If the container has overflow:visible, the element cannot overflow it.
762    if (containerOverflow.x == 'visible' && containerOverflow.y == 'visible') {
763      continue;
764    }
765
766    var containerRect = bot.dom.getClientRect(container);
767
768    // Zero-sized containers without overflow:visible hide all descendants.
769    if (containerRect.width == 0 || containerRect.height == 0) {
770      return bot.dom.OverflowState.HIDDEN;
771    }
772
773    // Check "underflow": if an element is to the left or above the container
774    var underflowsX = region.right < containerRect.left;
775    var underflowsY = region.bottom < containerRect.top;
776    if ((underflowsX && containerOverflow.x == 'hidden') ||
777        (underflowsY && containerOverflow.y == 'hidden')) {
778      return bot.dom.OverflowState.HIDDEN;
779    } else if ((underflowsX && containerOverflow.x != 'visible') ||
780               (underflowsY && containerOverflow.y != 'visible')) {
781      // When the element is positioned to the left or above a container, we
782      // have to distinguish between the element being completely outside the
783      // container and merely scrolled out of view within the container.
784      var containerScroll = getScroll(container);
785      var unscrollableX = region.right < containerRect.left - containerScroll.x;
786      var unscrollableY = region.bottom < containerRect.top - containerScroll.y;
787      if ((unscrollableX && containerOverflow.x != 'visible') ||
788          (unscrollableY && containerOverflow.x != 'visible')) {
789        return bot.dom.OverflowState.HIDDEN;
790      }
791      var containerState = bot.dom.getOverflowState(container);
792      return containerState == bot.dom.OverflowState.HIDDEN ?
793          bot.dom.OverflowState.HIDDEN : bot.dom.OverflowState.SCROLL;
794    }
795
796    // Check "overflow": if an element is to the right or below a container
797    var overflowsX = region.left >= containerRect.left + containerRect.width;
798    var overflowsY = region.top >= containerRect.top + containerRect.height;
799    if ((overflowsX && containerOverflow.x == 'hidden') ||
800        (overflowsY && containerOverflow.y == 'hidden')) {
801      return bot.dom.OverflowState.HIDDEN;
802    } else if ((overflowsX && containerOverflow.x != 'visible') ||
803               (overflowsY && containerOverflow.y != 'visible')) {
804      // If the element has fixed position and falls outside the scrollable area
805      // of the document, then it is hidden.
806      if (treatAsFixedPosition) {
807        var docScroll = getScroll(container);
808        if ((region.left >= htmlElem.scrollWidth - docScroll.x) ||
809            (region.right >= htmlElem.scrollHeight - docScroll.y)) {
810          return bot.dom.OverflowState.HIDDEN;
811        }
812      }
813      // If the element can be scrolled into view of the parent, it has a scroll
814      // state; unless the parent itself is entirely hidden by overflow, in
815      // which it is also hidden by overflow.
816      var containerState = bot.dom.getOverflowState(container);
817      return containerState == bot.dom.OverflowState.HIDDEN ?
818          bot.dom.OverflowState.HIDDEN : bot.dom.OverflowState.SCROLL;
819    }
820  }
821
822  // Does not overflow any ancestor.
823  return bot.dom.OverflowState.NONE;
824};

838/**
839 * Gets the client rectangle of the DOM element. It often returns the same value
840 * as Element.getBoundingClientRect, but is "fixed" for various scenarios:
841 * 1. Like goog.style.getClientPosition, it adjusts for the inset border in IE.
842 * 2. Gets a rect for <map>'s and <area>'s relative to the image using them.
843 * 3. Gets a rect for SVG elements representing their true bounding box.
844 * 4. Defines the client rect of the <html> element to be the window viewport.
845 *
846 * @param {!Element} elem The element to use.
847 * @return {!goog.math.Rect} The interaction box of the element.
848 */
849bot.dom.getClientRect = function(elem) {
850  var imageMap = bot.dom.maybeFindImageMap_(elem);
851  if (imageMap) {
852    return imageMap.rect;
853  } else if (bot.dom.isElement(elem, goog.dom.TagName.HTML)) {
854    // Define the client rect of the <html> element to be the viewport.
855    var doc = goog.dom.getOwnerDocument(elem);
856    var viewportSize = goog.dom.getViewportSize(goog.dom.getWindow(doc));
857    return new goog.math.Rect(0, 0, viewportSize.width, viewportSize.height);
858  } else {
859    var nativeRect;
860    try {
861      // TODO: in IE and Firefox, getBoundingClientRect includes stroke width,
862      // but getBBox does not.
863      nativeRect = elem.getBoundingClientRect();
864    } catch (e) {
865      // On IE < 9, calling getBoundingClientRect on an orphan element raises
866      // an "Unspecified Error". All other browsers return zeros.
867      return new goog.math.Rect(0, 0, 0, 0);
868    }
869
870    var left = nativeRect.left;
871    var right = nativeRect.right;
872    var top = nativeRect.top;
873    var bottom = nativeRect.bottom;
874
875    var rect = new goog.math.Rect(left, top, right - left, bottom - top);
876
877    // In IE, the element can additionally be offset by a border around the
878    // documentElement or body element that we have to subtract.
879    if (goog.userAgent.IE && elem.ownerDocument.body) {
880      var doc = goog.dom.getOwnerDocument(elem);
881      rect.left -= doc.documentElement.clientLeft + doc.body.clientLeft;
882      rect.top -= doc.documentElement.clientTop + doc.body.clientTop;
883    }
884
885    // Opera sometimes falsely report zero size bounding rects.
886    if (goog.userAgent.OPERA) {
887      if (rect.width == 0 && elem.offsetWidth > 0) {
888        rect.width = elem.offsetWidth;
889      }
890      if (rect.height == 0 && elem.offsetHeight > 0) {
891        rect.height = elem.offsetHeight;
892      }
893    }
894
895    // On Gecko < 12, getBoundingClientRect does not account for CSS transforms.
896    // TODO: Remove this when we drop support for FF3.6 and FF10.
897    if (goog.userAgent.GECKO && !bot.userAgent.isEngineVersion(12)) {
898      transformLegacyFirefoxClientRect(elem);
899    }
900
901    return rect;
902  }
903
904  function transformLegacyFirefoxClientRect(container) {
905    var win = goog.dom.getWindow(goog.dom.getOwnerDocument(container));
906    var transform = win.getComputedStyle(container, null)['MozTransform'];
907    var matches = transform.match(bot.dom.CSS_TRANSFORM_MATRIX_REGEX_);
908
909    if (matches) {
910      var a = parseFloat(matches[1]), b = parseFloat(matches[2]),
911          c = parseFloat(matches[3]), d = parseFloat(matches[4]),
912          x = parseFloat(matches[5]), y = parseFloat(matches[6]);
913      var right = rect.left + rect.width, bottom = rect.top + rect.height;
914      var leftXa = rect.left * a, rightXa = right * a,
915          leftXb = rect.left * b, rightXb = right * b,
916          topXc = rect.top * c, bottomXc = bottom * c,
917          topXd = rect.top * d, bottomXd = bottom * d;
918      var topLeftX = leftXa + topXc + x,
919          topLeftY = leftXb + topXd + y,
920          topRightX = rightXa + topXc + x,
921          topRightY = rightXb + topXd + y,
922          bottomLeftX = leftXa + bottomXc + x,
923          bottomLeftY = leftXb + bottomXd + y,
924          bottomRightX = rightXa + bottomXc + x,
925          bottomRightY = rightXb + bottomXd + y;
926      rect.left = Math.min(topLeftX, topRightX, bottomLeftX, bottomRightX);
927      rect.top = Math.min(topLeftY, topRightY, bottomLeftY, bottomRightY);
928      var newRight = Math.max(topLeftX, topRightX, bottomLeftX, bottomRightX);
929      var newBottom = Math.max(topLeftY, topRightY, bottomLeftY, bottomRightY);
930      rect.width = newRight - rect.left;
931      rect.height = newBottom - rect.top;
932    }
933
934    // The computed transform style not not take into account parent transforms.
935    var parentContainer = bot.dom.getParentElement(container);
936    if (parentContainer) {
937      transformLegacyFirefoxClientRect(parentContainer);
938    }
939  }
940};

943/**
944 * If given a <map> or <area> element, finds the corresponding image and client
945 * rectangle of the element; otherwise returns null. The return value is an
946 * object with 'image' and 'rect' properties. When no image uses the given
947 * element, the returned rectangle is present but has zero size.
948 *
949 * @param {!Element} elem Element to test.
950 * @return {?{image: Element, rect: !goog.math.Rect}} Image and rectangle.
951 * @private
952 */
953bot.dom.maybeFindImageMap_ = function(elem) {
954  // If not a <map> or <area>, return null indicating so.
955  var isMap = bot.dom.isElement(elem, goog.dom.TagName.MAP);
956  if (!isMap && !bot.dom.isElement(elem, goog.dom.TagName.AREA)) {
957    return null;
958  }
959
960  // Get the <map> associated with this element, or null if none.
961  var map = isMap ? elem :
962      (bot.dom.isElement(elem.parentNode, goog.dom.TagName.MAP) ?
963          elem.parentNode : null);
964
965  var image = null, rect = null;
966  if (map && map.name) {
967    var mapDoc = goog.dom.getOwnerDocument(map);
968
969    // The "//*" XPath syntax can confuse the closure compiler, so we use
970    // the "/descendant::*" syntax instead.
971    // TODO: Try to find a reproducible case for the compiler bug.
972    // TODO: Restrict to applet, img, input:image, and object nodes.
973    var imageXpath = '/descendant::*[@usemap = "#' + map.name + '"]';
974
975    // TODO: Break dependency of bot.locators on bot.dom,
976    // so bot.locators.findElement can be called here instead.
977    image = bot.locators.xpath.single(imageXpath, mapDoc);
978
979    if (image) {
980      rect = bot.dom.getClientRect(image);
981      if (!isMap && elem.shape.toLowerCase() != 'default') {
982        // Shift and crop the relative area rectangle to the map.
983        var relRect = bot.dom.getAreaRelativeRect_(elem);
984        var relX = Math.min(Math.max(relRect.left, 0), rect.width);
985        var relY = Math.min(Math.max(relRect.top, 0), rect.height);
986        var w = Math.min(relRect.width, rect.width - relX);
987        var h = Math.min(relRect.height, rect.height - relY);
988        rect = new goog.math.Rect(relX + rect.left, relY + rect.top, w, h);
989      }
990    }
991  }
992
993  return {image: image, rect: rect || new goog.math.Rect(0, 0, 0, 0)};
994};

997/**
998 * Returns the bounding box around an <area> element relative to its enclosing
999 * <map>. Does not apply to <area> elements with shape=='default'.
1000 *
1001 * @param {!Element} area Area element.
1002 * @return {!goog.math.Rect} Bounding box of the area element.
1003 * @private
1004 */
1005bot.dom.getAreaRelativeRect_ = function(area) {
1006  var shape = area.shape.toLowerCase();
1007  var coords = area.coords.split(',');
1008  if (shape == 'rect' && coords.length == 4) {
1009    var x = coords[0], y = coords[1];
1010    return new goog.math.Rect(x, y, coords[2] - x, coords[3] - y);
1011  } else if (shape == 'circle' && coords.length == 3) {
1012    var centerX = coords[0], centerY = coords[1], radius = coords[2];
1013    return new goog.math.Rect(centerX - radius, centerY - radius,
1014                              2 * radius, 2 * radius);
1015  } else if (shape == 'poly' && coords.length > 2) {
1016    var minX = coords[0], minY = coords[1], maxX = minX, maxY = minY;
1017    for (var i = 2; i + 1 < coords.length; i += 2) {
1018      minX = Math.min(minX, coords[i]);
1019      maxX = Math.max(maxX, coords[i]);
1020      minY = Math.min(minY, coords[i + 1]);
1021      maxY = Math.max(maxY, coords[i + 1]);
1022    }
1023    return new goog.math.Rect(minX, minY, maxX - minX, maxY - minY);
1024  }
1025  return new goog.math.Rect(0, 0, 0, 0);
1026};

1029/**
1030 * Gets the element's client rectangle as a box, optionally clipped to the
1031 * given coordinate or rectangle relative to the client's position. A coordinate
1032 * is treated as a 1x1 rectangle whose top-left corner is the coordinate.
1033 *
1034 * @param {!Element} elem The element.
1035 * @param {!(goog.math.Coordinate|goog.math.Rect)=} opt_region
1036 *     Coordinate or rectangle relative to the top-left corner of the element.
1037 * @return {!goog.math.Box} The client region box.
1038 */
1039bot.dom.getClientRegion = function(elem, opt_region) {
1040  var region = bot.dom.getClientRect(elem).toBox();
1041
1042  if (opt_region) {
1043    var rect = opt_region instanceof goog.math.Rect ? opt_region :
1044        new goog.math.Rect(opt_region.x, opt_region.y, 1, 1);
1045    region.left = goog.math.clamp(
1046        region.left + rect.left, region.left, region.right);
1047    region.top = goog.math.clamp(
1048        region.top + rect.top, region.top, region.bottom);
1049    region.right = goog.math.clamp(
1050        region.left + rect.width, region.left, region.right);
1051    region.bottom = goog.math.clamp(
1052        region.top + rect.height, region.top, region.bottom);
1053  }
1054
1055  return region;

1091/**
1092 * @param {!Element} elem Element.
1093 * @param {!Array.<string>} lines Accumulated visible lines of text.
1094 * @private
1095 */
1096bot.dom.appendVisibleTextLinesFromElement_ = function(elem, lines) {
1097  function currLine() {
1098    return /** @type {string|undefined} */ (goog.array.peek(lines)) || '';
1099  }
1100
1101  // TODO: Add case here for textual form elements.
1102  if (bot.dom.isElement(elem, goog.dom.TagName.BR)) {
1103    lines.push('');
1104  } else {
1105    // TODO: properly handle display:run-in
1106    var isTD = bot.dom.isElement(elem, goog.dom.TagName.TD);
1107    var display = bot.dom.getEffectiveStyle(elem, 'display');
1108    // On some browsers, table cells incorrectly show up with block styles.
1109    var isBlock = !isTD &&
1110        !goog.array.contains(bot.dom.INLINE_DISPLAY_BOXES_, display);
1111
1112    // Add a newline before block elems when there is text on the current line,
1113    // except when the previous sibling has a display: run-in.
1114    // Also, do not run-in the previous sibling if this element is floated.
1115
1116    var previousElementSibling = goog.dom.getPreviousElementSibling(elem);
1117    var prevDisplay = (previousElementSibling) ?
1118        bot.dom.getEffectiveStyle(previousElementSibling, 'display') : '';
1119    // TODO: getEffectiveStyle should mask this for us
1120    var thisFloat = bot.dom.getEffectiveStyle(elem, 'float') ||
1121        bot.dom.getEffectiveStyle(elem, 'cssFloat') ||
1122        bot.dom.getEffectiveStyle(elem, 'styleFloat');
1123    var runIntoThis = prevDisplay == 'run-in' && thisFloat == 'none';
1124    if (isBlock && !runIntoThis && !goog.string.isEmpty(currLine())) {
1125      lines.push('');
1126    }
1127
1128    // This element may be considered unshown, but have a child that is
1129    // explicitly shown (e.g. this element has "visibility:hidden").
1130    // Nevertheless, any text nodes that are direct descendants of this
1131    // element will not contribute to the visible text.
1132    var shown = bot.dom.isShown(elem);
1133
1134    // All text nodes that are children of this element need to know the
1135    // effective "white-space" and "text-transform" styles to properly
1136    // compute their contribution to visible text. Compute these values once.
1137    var whitespace = null, textTransform = null;
1138    if (shown) {
1139      whitespace = bot.dom.getEffectiveStyle(elem, 'white-space');
1140      textTransform = bot.dom.getEffectiveStyle(elem, 'text-transform');
1141    }
1142
1143    goog.array.forEach(elem.childNodes, function(node) {
1144      if (node.nodeType == goog.dom.NodeType.TEXT && shown) {
1145        var textNode = /** @type {!Text} */ (node);
1146        bot.dom.appendVisibleTextLinesFromTextNode_(textNode, lines,
1147            whitespace, textTransform);
1148      } else if (bot.dom.isElement(node)) {
1149        var castElem = /** @type {!Element} */ (node);
1150        bot.dom.appendVisibleTextLinesFromElement_(castElem, lines);
1151      }
1152    });
1153
1154    var line = currLine();
1155
1156    // Here we differ from standard innerText implementations (if there were
1157    // such a thing). Usually, table cells are separated by a tab, but we
1158    // normalize tabs into single spaces.
1159    if ((isTD || display == 'table-cell') && line &&
1160        !goog.string.endsWith(line, ' ')) {
1161      lines[lines.length - 1] += ' ';
1162    }
1163
1164    // Add a newline after block elems when there is text on the current line,
1165    // and the current element isn't marked as run-in.
1166    if (isBlock && display != 'run-in' && !goog.string.isEmpty(line)) {
1167      lines.push('');
1168    }
1169  }
1170};

1190/**
1191 * @param {!Text} textNode Text node.
1192 * @param {!Array.<string>} lines Accumulated visible lines of text.
1193 * @param {?string} whitespace Parent element's "white-space" style.
1194 * @param {?string} textTransform Parent element's "text-transform" style.
1195 * @private
1196 */
1197bot.dom.appendVisibleTextLinesFromTextNode_ = function(textNode, lines,
1198    whitespace, textTransform) {
1199  // First, remove zero-width characters. Do this before regularizing spaces as
1200  // the zero-width space is both zero-width and a space, but we do not want to
1201  // make it visible by converting it to a regular space.
1202  // The replaced characters are:
1203  //   U+200B: Zero-width space
1204  //   U+200E: Left-to-right mark
1205  //   U+200F: Right-to-left mark
1206  var text = textNode.nodeValue.replace(/[\u200b\u200e\u200f]/g, '');
1207
1208  // Canonicalize the new lines, and then collapse new lines
1209  // for the whitespace styles that collapse. See:
1210  // https://developer.mozilla.org/en/CSS/white-space
1211  text = goog.string.canonicalizeNewlines(text);
1212  if (whitespace == 'normal' || whitespace == 'nowrap') {

1217  // non-breaking, otherwise, collapse all breaking spaces. Breaking spaces are

1223  }

1226    text = text.replace(/(^|\s)(\S)/g, function() {
1227      return arguments[1] + arguments[2].toUpperCase();

1229  } else if (textTransform == 'uppercase') {
1230    text = text.toUpperCase();
1231  } else if (textTransform == 'lowercase') {

1233  }
1234

1236  if (goog.string.endsWith(currLine, ' ') &&
1237      goog.string.startsWith(text, ' ')) {
1238    text = text.substr(1);
1239  }
1240  lines.push(currLine + text);
