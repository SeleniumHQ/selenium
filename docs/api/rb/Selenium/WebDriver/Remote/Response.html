def error
  klass = Error.for_code(@payload['status']) || return

  ex = klass.new(error_message)
  ex.set_backtrace(caller)
  add_backtrace ex

  ex
end

def error_message
  val = @payload['value']
  msg = val['message'] or return ""

  msg << " (#{ val['class'] })" if val['class']

  msg
end
