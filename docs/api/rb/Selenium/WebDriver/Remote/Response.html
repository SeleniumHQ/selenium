def error
  klass = Error.for_code(status) || return

  ex = klass.new(error_message)
  ex.set_backtrace(caller)
  add_backtrace ex

  ex
end

def error_message
  val = value

  case val
  when Hash
    msg = val['message'] or return "unknown error"
    msg << " (#{ val['class'] })" if val['class']
  when String
    msg = val
  else
    msg = "unknown error, status=#{status}: #{val.inspect}"
  end

  msg
end
