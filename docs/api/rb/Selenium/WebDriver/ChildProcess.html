def initialize(*args)
  @args = args

  if Platform.jruby?
    extend JRubyProcess
  elsif Platform.ironruby?
    extend IronRubyProcess
  elsif Platform.os == :windows
    extend WindowsProcess
  end
end

def start
  @pid = fork do
    unless $DEBUG
      [STDOUT, STDERR].each { |io| io.reopen("/dev/null") }
    end

    exec(*@args)
  end

  self
end

def ugly_death?
  code = exit_value()
  # if exit_val is nil, the process is still alive
  code && code != 0
end
