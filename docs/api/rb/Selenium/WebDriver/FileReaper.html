def reap(file)
  return unless reap?

  raise Error::WebDriverError, "file not added for reaping: #{file.inspect}" unless tmp_files.include?(file)

  FileUtils.rm_rf tmp_files.delete(file)
end

def reap!
  if reap?
    tmp_files.each { |file| FileUtils.rm_rf(file) }
    true
  else
    false
  end
end
