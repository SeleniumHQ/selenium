def for(browser, *args)
  bridge = case browser
           when :ie, :internet_explorer
             IE::Bridge.new(*args)
           when :remote
             Remote::Bridge.new(*args)
           when :chrome
             Chrome::Bridge.new(*args)
           when :firefox, :ff
             Firefox::Bridge.new(*args)
           else
             raise ArgumentError, "unknown driver: #{browser.inspect}"
           end

   new(bridge)
end
