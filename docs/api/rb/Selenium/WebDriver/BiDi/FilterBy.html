def self.log_level(level = nil)
  unless %w[debug error info warning].include?(level)
    raise Error::WebDriverError,
          "Valid log levels are 'debug', 'error', 'info' and 'warning'. Received: #{level}"
  end
  FilterBy.new(level)
end
